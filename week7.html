Fish Playground — Turbine

Pseudocode:

START
  read light sensor value
  IF value is low 
    THEN rotate servo to 180°
  bright 
    THEN rotate servo to 0°
  repeat forever

Arduino Code:

// Fish Playground — Turbine
// Servo rotates based on light sensor input
// Source: class workshop W7

#include <Servo.h>
int photo;    // stores light sensor reading
Servo s;      // servo object

void setup() {
  s.attach(9);        // servo on pin 9
  Serial.begin(9600); // for monitoring values
}

void loop() {
  photo = analogRead(A1);   // read light sensor
  Serial.println(photo);    // print to monitor (for debugging)

  if (photo < 700) {        // if dark/covered
    s.write(180);           // spin one direction
  } else {                  // if bright
    s.write(0);             // spin other direction
  }
}

Shadow Theremin — Light Sensor + Sound

Pseudocode:

START
  calibrate sensor for 5 seconds
    record highest and lowest light values seen
  
  LOOP forever:
    read current light value
    map value to a pitch range
    play that pitch for 20ms
    wait 10ms

Arduino Code:

// Shadow Theremin — LDR controls sound pitch
// Source: class workshop W7
// Note: the black component with shiny bottom = the LDR sensor

int sensorValue;
int sensorLow = 1023;   // will store calibrated min
int sensorHigh = 0;     // will store calibrated max
const int ledPin = 13;  // LED signals calibration end

void setup() {
  pinMode(ledPin, OUTPUT);
  digitalWrite(ledPin, HIGH); // LED on = calibrating

  // calibrate for 5 seconds
  while (millis() < 5000) {
    sensorValue = analogRead(A0);
    if (sensorValue > sensorHigh) { sensorHigh = sensorValue; }
    if (sensorValue < sensorLow)  { sensorLow  = sensorValue; }
  }
  digitalWrite(ledPin, LOW); // LED off = ready
}

void loop() {
  sensorValue = analogRead(A0);
  // map sensor range to pitch range
  int pitch = map(sensorValue, sensorLow, sensorHigh, 50, 4000);
  tone(8, pitch, 20); // play pitch on pin 8 for 20ms
  delay(10);
}

Headspace — Ultrasonic Obstacle Sensor

Pseudocode:

START
  LOOP forever:
    send ultrasonic pulse from TRIG pin
    measure time for echo to return
    calculate distance in cm
    
    IF distance less than 50cm
      THEN turn buzzer ON
    ELSE
      turn buzzer OFF
    
    print distance to serial monitor
    wait 500ms

Arduino Code:

// Headspace — Ultrasonic Distance Sensor + Buzzer
// Source: class workshop W7
// We put bags over our heads and walked around to test this one

const int TRIG_PIN   = 7;
const int ECHO_PIN   = 6;
const int BUZZER_PIN = 3;
const int DISTANCE_THRESHOLD = 50; // cm

float duration_us, distance_cm;

void setup() {
  Serial.begin(9600);
  pinMode(TRIG_PIN,   OUTPUT);
  pinMode(ECHO_PIN,   INPUT);
  pinMode(BUZZER_PIN, OUTPUT);
}

void loop() {
  // send pulse
  digitalWrite(TRIG_PIN, HIGH);
  delayMicroseconds(10);
  digitalWrite(TRIG_PIN, LOW);

  // measure echo
  duration_us = pulseIn(ECHO_PIN, HIGH);
  distance_cm = 0.017 * duration_us;

  // activate buzzer if close
  if (distance_cm < DISTANCE_THRESHOLD)
    digitalWrite(BUZZER_PIN, HIGH);
  else
    digitalWrite(BUZZER_PIN, LOW);

  Serial.print("distance: ");
  Serial.print(distance_cm);
  Serial.println(" cm");
  delay(500);
}
